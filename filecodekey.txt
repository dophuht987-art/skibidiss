-- 1️⃣ KHAI BÁO KEY & FILE LƯU TRỮ
local key = "z" 
local keyLink = "https://www.tiktok.com/@b.yt03"
local fileName = "SavedKey_MyScript.txt" -- Tên file sẽ lưu trong thư mục workspace

-- 2️⃣ HÀM LƯU VÀ ĐỌC KEY
local function saveKey(keyValue)
    writefile(fileName, keyValue)
end

local function getSavedKey()
    if isfile(fileName) then
        return readfile(fileName)
    end
    return ""
end

-- 3️⃣ KIỂM TRA KEY TỰ ĐỘNG KHI MỞ SCRIPT
local saved = getSavedKey()
local keyOk = false

if saved == key then
    keyOk = true
    game.StarterGui:SetCore("SendNotification", {
        Title = "Tự động đăng nhập ⚡",
        Text = "Chào mừng bạn quay trở lại!",
        Duration = 3;
    })
end

-- 4️⃣ GIAO DIỆN NHẬP KEY (Chỉ hiện nếu chưa có key đúng)
if not keyOk then
    local CoreGui = game:GetService("CoreGui")
    local ScreenGui = Instance.new("ScreenGui")
    local Frame = Instance.new("Frame")
    local TextBox = Instance.new("TextBox")
    local CheckButton = Instance.new("TextButton")
    local GetKeyButton = Instance.new("TextButton")
    local UICorner = Instance.new("UICorner")

    ScreenGui.Name = "KeySystem"
    ScreenGui.Parent = CoreGui
    
    -- (Phần code giao diện giữ nguyên như của bạn, tôi tóm gọn lại cho sạch)
    Frame.Parent = ScreenGui
    Frame.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
    Frame.Size = UDim2.new(0, 300, 0, 160)
    Frame.Position = UDim2.new(0.5, -150, 0.5, -80)
    UICorner.Parent = Frame

    TextBox.Parent = Frame
    TextBox.PlaceholderText = "Nhập key tại đây 🍎"
    TextBox.Text = saved -- Tự động điền key cũ vào đây nếu có
    TextBox.Size = UDim2.new(0.9, 0, 0, 35)
    TextBox.Position = UDim2.new(0.05, 0, 0.25, 0)
    TextBox.BackgroundColor3 = Color3.fromRGB(70, 70, 70)
    TextBox.TextColor3 = Color3.fromRGB(255, 255, 255)
    Instance.new("UICorner", TextBox)

    CheckButton.Parent = Frame
    CheckButton.Text = "Kiểm tra key ✅"
    CheckButton.Size = UDim2.new(0.4, 0, 0, 30)
    CheckButton.Position = UDim2.new(0.05, 0, 0.65, 0)
    CheckButton.BackgroundColor3 = Color3.fromRGB(80, 170, 80)
    Instance.new("UICorner", CheckButton)

    GetKeyButton.Parent = Frame
    GetKeyButton.Text = "Lấy key 🔑"
    GetKeyButton.Size = UDim2.new(0.4, 0, 0, 30)
    GetKeyButton.Position = UDim2.new(0.55, 0, 0.65, 0)
    GetKeyButton.BackgroundColor3 = Color3.fromRGB(100, 100, 255)
    Instance.new("UICorner", GetKeyButton)

    GetKeyButton.MouseButton1Click:Connect(function()
        setclipboard(keyLink)
        game.StarterGui:SetCore("SendNotification", {Title = "Đã copy link!"; Text = "Dán vào trình duyệt nhé!"; Duration = 4})
    end)

    CheckButton.MouseButton1Click:Connect(function()
        if TextBox.Text == key then
            saveKey(TextBox.Text) -- LƯU KEY VÀO FILE KHI NHẬP ĐÚNG
            keyOk = true
            ScreenGui:Destroy()
            game.StarterGui:SetCore("SendNotification", {Title = "Thành công!"; Text = "Đã lưu key cho lần sau."; Duration = 3})
        else
            game.StarterGui:SetCore("SendNotification", {Title = "Sai key!"; Text = "Vui lòng kiểm tra lại."; Duration = 3})
        end
    end)
    
    -- Nút X để tắt
    local CloseButton = Instance.new("TextButton", Frame)
    CloseButton.Text = "X"
    CloseButton.Size = UDim2.new(0, 25, 0, 25)
    CloseButton.Position = UDim2.new(1, -30, 0, 5)
    CloseButton.BackgroundColor3 = Color3.fromRGB(255, 60, 60)
    CloseButton.MouseButton1Click:Connect(function() ScreenGui:Destroy() end)
    Instance.new("UICorner", CloseButton)
end

-- 5️⃣ CHỜ ĐỢI VÀ CHẠY SCRIPT CHÍNH
repeat task.wait() until keyOk == true

print("Script đã chạy thành công!")
